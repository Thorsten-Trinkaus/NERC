import json
from collections import Counter
import pandas as pd

def load_json_dataset(path):
    data = []
    with open(path) as f:
        for line in f:
            data.append(json.loads(line))
    return data

def extract_labels(dataset):
    labels = []
    for item in dataset:
        if "labels" in item:
            labels.extend(item["labels"])
        elif "y_str" in item:
            labels.extend(item["y_str"])
    return labels

def label_frequency(labels):
    counter = Counter(labels)
    df = pd.DataFrame(counter.items(), columns=["label", "frequency"])
    return df.sort_values("frequency", ascending=False)
